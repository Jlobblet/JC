239 void Vector_filter(Vector* vec, vector_filter_t predicate, ...) {
240  va_list args;
241  va_start(args, predicate);
242  for (uptr i = 0; i < vec->length; i++) {
243  if (predicate(vec->data[i], args)) { continue; }
244  // Shift all elements after `i` to the left.
245  // This is done in reverse order to avoid shifting elements that are not yet visited.
246  // This removes the ith element.
247  memcpy(vec->data + i, vec->data + i + 1, (vec->length - i - 1) * sizeof(void*));
248  // Decrement the length.
249  vec->length--;
250  // Decrement `i` to account for the shift.
251  i--;
252  }
253  va_end(args);
254 }
